133 if (j.contains("tile_properties"))

141 // Helper map for string to enum conversion
142 std::unordered_map<std::string, ObstacleType> stringToObstacleType = {
143 {"Wall", ObstacleType::Wall},
144 {"Water", ObstacleType::Water},
145 {"Forest", ObstacleType::Forest},
146 {"Grass", ObstacleType::Grass},
147 {"Path", ObstacleType::Path}
148 };
149
150 // 2. Process all tiles to set their type and cost
151 if (j.contains("tiles")) {

153 int x = tile.value("x", -1);

155 std::string typeStr = tile.value("type", "None");

157 // Use this->at() or just at() - no more "grid."
158 if (x >= 0 && x < getWidth() && y >= 0 && y < getHeight())
159 {
160 auto it = stringToObstacleType.find(typeStr);
161 at(x, y).obstacle = (it != stringToObstacleType.end()) ? it->second : ObstacleType::None;
162 at(x, y).cost = tileCosts.count(typeStr) ? tileCosts[typeStr] : 1;
163 }
164 }
165 }
166
167 // 3. Process the zones
168 if (j.contains("zones")) {
169 for (auto& [name, zone_data] : j["zones"].items()) {
