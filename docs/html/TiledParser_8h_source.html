53 static bool loadTiledMap(const std::string& filename, Grid& grid) {
54 std::ifstream file(filename);
55 if (!file) {
56 std::cerr << "Failed to open Tiled file: " << filename << std::endl;
57 return false;
58 }
59
60 nlohmann::json j;
61 try {
62 j = nlohmann::json::parse(file);
63 } catch (const nlohmann::json::parse_error& e) {
64 std::cerr << "Error parsing Tiled JSON: " << e.what() << std::endl;
65 return false;
66 }
67
68 // Validate this is a Tiled map
69 if (!j.contains("type") || j["type"] != "map") {
70 std::cerr << "Not a valid Tiled map file" << std::endl;
71 return false;
72 }
73
74 // Get map dimensions
75 int mapWidth = j.value("width", 0);
76 int mapHeight = j.value("height", 0);
77 int tileWidth = j.value("tilewidth", 32);
78 int tileHeight = j.value("tileheight", 32);
79
80 std::cout << "Loading Tiled map: " << mapWidth << "x" << mapHeight
81 << " (tile size: " << tileWidth << "x" << tileHeight << ")" << std::endl;
82
83 // Resize grid to match map
84 grid = Grid(mapWidth, mapHeight);
85
86 // Load tilesets
87 std::vector<TiledTileset> tilesets;
88 if (j.contains("tilesets")) {
89 for (const auto& tilesetRef : j["tilesets"]) {
90 TiledTileset tileset;
91 tileset.firstgid = tilesetRef.value("firstgid", 1);
92
93 // Handle embedded vs external tilesets
94 if (tilesetRef.contains("source")) {
95 // External tileset - would need to load from separate file
96 std::cerr << "External tilesets not yet supported" << std::endl;
97 continue;
98 } else {
99 // Embedded tileset
100 tileset.name = tilesetRef.value("name", "");
101 tileset.tilewidth = tilesetRef.value("tilewidth", tileWidth);
102 tileset.tileheight = tilesetRef.value("tileheight", tileHeight);
103 tileset.tilecount = tilesetRef.value("tilecount", 0);
104 tileset.columns = tilesetRef.value("columns", 1);
105 tileset.image = tilesetRef.value("image", "");
106 }
107
108 // Load tile properties and types
109 if (tilesetRef.contains("tiles")) {
110 for (const auto& tileData : tilesetRef["tiles"]) {
111 int tileId = tileData.value("id", 0);
112 int globalId = tileset.firstgid + tileId;
113
114 if (tileData.contains("type")) {
115 tileset.tileTypes[globalId] = tileData["type"];
116 }
117
118 if (tileData.contains("properties")) {
119 for (const auto& prop : tileData["properties"]) {
120 if (prop.contains("name") && prop["name"] == "cost") {
121 tileset.tileCosts[globalId] = prop.value("value", 1);
122 }
123 }
124 }
125 }
126 }
127
128 tilesets.push_back(tileset);
129 std::cout << "Loaded tileset: " << tileset.name << " (firstgid=" << tileset.firstgid << ")" << std::endl;
130 }
131 }
132
133 // Load layers
134 if (j.contains("layers")) {
135 for (const auto& layerJson : j["layers"]) {
136 std::string layerType = layerJson.value("type", "");
137 std::string layerName = layerJson.value("name", "");
138 bool visible = layerJson.value("visible", true);
139
140 if (!visible) continue; // Skip invisible layers
141
142 if (layerType == "tilelayer") {
143 // Tile layer - contains the actual tile data
144 processTileLayer(layerJson, grid, tilesets);
145 } else if (layerType == "objectgroup") {
146 // Object layer - contains zones, spawn points, etc.
147 processObjectGroup(layerJson, grid);
148 }
149 }
150 }

152 return true;

156 static void processTileLayer(const nlohmann::json& layer, Grid& grid, const std::vector<TiledTileset>& tilesets) {
157 std::string layerName = layer.value("name", "");
158 int layerWidth = layer.value("width", 0);
159 int layerHeight = layer.value("height", 0);
160
161 std::cout << "Processing tile layer: " << layerName << std::endl;
162
163 if (layer.contains("data")) {
164 std::vector<int> tileData = layer["data"];
165
166 for (int y = 0; y < layerHeight && y < grid.getHeight(); ++y) {
167 for (int x = 0; x < layerWidth && x < grid.getWidth(); ++x) {
168 int index = y * layerWidth + x;
169 if (index >= tileData.size()) continue;
170
171 int gid = tileData[index];
172 if (gid == 0) continue; // Empty tile
173
174 // Find which tileset this tile belongs to
175 const TiledTileset* tileset = nullptr;
176 for (auto it = tilesets.rbegin(); it != tilesets.rend(); ++it) {
177 if (gid >= it->firstgid) {
178 tileset = &(*it);
179 break;
180 }
181 }
182
183 if (!tileset) continue;
184
185 // Set tile properties
186 Cell& cell = grid.at(x, y);
187
188 // Set obstacle type based on tile type
189 if (tileset->tileTypes.count(gid)) {
190 std::string tileType = tileset->tileTypes.at(gid);
191 cell.obstacle = stringToObstacleType(tileType);
192 }
193
194 // Set movement cost
195 if (tileset->tileCosts.count(gid)) {
196 cell.cost = tileset->tileCosts.at(gid);
197 } else {
198 // Default costs based on obstacle type
199 cell.cost = getDefaultCost(cell.obstacle);
200 }
201 }
202 }
203 }
